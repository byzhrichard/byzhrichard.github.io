class B_Seq2Seq_Decoder(nn.Module):
    """
    Decoder:
        śĮŹś¨°ŤĺďŚÖ•šłÄšł™ tokenÔľĆŤĺďŚáļšłÄś≠•ťĘĄśĶč

    ŤĺďŚÖ•:
        y_t: (B,)                ŚĹďŚČćś≠•ŤĺďŚÖ• token id
        h_prev: (L, B, H)        šłäšłÄś≠•ťöźŤóŹÁä∂śÄĀ

    ŤĺďŚáļ:
        logits: (B, vocab_size)
        h_t: (L, B, H)
    """

    def __init__(
        self,
        vocab_size: int,
        emb_dim: int,
        hidden_size: int,
        num_layers: int = 1,
        pad_idx: int = 0, # ŤßĄŚģöŤĮćŤ°®ťáĆ index=0 šĽ£Ť°® <pad>
        dropout: float = 0.1
    ):
        super().__init__()
        # śää ‚ÄúÁ¶Ľśē£ÁöĄ token id‚ÄĚ śė†ŚįĄśąź ‚ÄúŤŅěÁĽ≠ÁöĄŚźĎťáŹŤ°®Á§ļÔľąembeddingÔľČ‚ÄĚ
        self.embedding = nn.Embedding(vocab_size, emb_dim, padding_idx=pad_idx)
        # GRUśúÄÁúĀšļč, šĻüŚŹĮšĽ•śćĘśąźŚąęÁöĄ
        self.gru = nn.GRU(
            input_size=emb_dim,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0.0
        )
        self.fc = nn.Linear(hidden_size, vocab_size)

    def forward(self, y_t, h_prev):
        """
        ŤĺďŚÖ•:
            y_t: (B,)                ŚĹďŚČćś≠•ŤĺďŚÖ• token id
            h_prev: (L, B, H)        šłäšłÄś≠•ťöźŤóŹÁä∂śÄĀÔľąLŚĪāÔľĆśĮŹŚĪāšłÄšł™ťöźŤóŹÁä∂śÄĀÔľČ

        device = src.device

                outputs.append(logits)

                # teacher forcing: šłčšłÄšł™ŤĺďŚÖ•ÁĒ®ÁúüŚģě token
                dec_input = tgt[:, t] if teacher_force else top1

            for _ in range(max_len):

                outputs.append(logits)

                dec_input = top1

            outputs = torch.stack(outputs, dim=1)  # (B, max_len, vocab_size)
