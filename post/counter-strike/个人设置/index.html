autoexec.cfg

D:\SteamLibrary\steamapps\common\Counter-Strike Global Offensive\game\csgo\cfg\autoexec.cfg

  • binddefaults: 还原所有按键绑定
  • key_listboundkeys: 显示所有按键绑定
  • exec autoexec.cfg: 在局内控制台使用, 可以自动使用更新后的cfg
// 鼠标灵敏度
sensitivity 1.20
zoom_sensitivity_ratio 1

// 按键绑定
bind "z" "slot10" // 火                
bind "c" "slot8" // 烟                   
bind "v" "slot3;slot7" // 闪               
bind "x" "slot6" // 雷 
bind "q" "slot11" // 电击枪

bind "mouse5" "+voicerecord" // 麦克风
bind "capslock" "+sprint" // 静步
bind "shift" "+duck" // 蹲
bind "p" "toggleconsole" // 控制台
unbind "`"

unbind "u" 
bind "ENTER" "messagemode2" // 队内聊天
unbind "y"
bind "\" "messagemode" // 全局聊天

bind "-" "noclip"
bind "[" "say_team .clear"
bind "]" "say_team .rethrow" 

bind ","  "+radialradio"  // 快捷聊天轮盘 
bind "."  "+radialradio2" // 快捷聊天轮盘 
bind "/"  "+radialradio3" // 快捷聊天轮盘 

// F1和F2要留给投票

bind "F4" "say_team .drop"    // 发刀
bind "F5" "say_team .pause"   // 暂停
bind "F6" "say_team .unpause" // 取消暂停

unbind "F8"
