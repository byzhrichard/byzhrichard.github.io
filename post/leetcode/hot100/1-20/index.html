1. 两数之和 

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        for i in range(len(nums)):
            # 需要找到的目标数字
            res = target - nums[i]
            # 遍历剩下的元素，查找是否存在该数字
            if res in nums[i+1:]:
                # 还需加上i+1才是该元素在nums中的索引
                return [i, nums[i+1:].index(res) + i+1]

2. 两数相加 

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        n1 = self.get_num(l1)
        n2 = self.get_num(l2)
        n = n1 + n2

        result_node = ListNode()
        temp_node = result_node
        # while True + break的方式实现遍历链表!!!
        while True:
            # 实际工作
            temp_node.val = n % 10
            
            # 更新退出数据
            n = n // 10

            # 判断退出条件
            if n == 0:
                break
            # 不退出的话, 更新数据
            temp_node.next = ListNode()
            temp_node = temp_node.next
        
        return result_node
        
    def get_num(self, l):
        result = 0
        temp = 0
        while l is not None:
            result += l.val * (10**temp)
            temp += 1
            l = l.next
        return result

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    # 手动添加carry=0参数
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode], carry=0) -> Optional[ListNode]:
        # 递归边界
        if l1 is None and l2 is None and carry == 0:  
            return None

        # 实际工作
        s = carry
        if l1:
            s += l1.val
            l1 = l1.next
        if l2:
            s += l2.val
            l2 = l2.next

        # s 除以 10 的余数为当前节点值，商为进位
        # (若题目要返回链表, 可考虑递归, 返回值是首个节点!!!)
        return ListNode(s % 10, self.addTwoNumbers(l1, l2, s // 10))

3. 无重复字符的最长子串 

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        # 本质：end始终向后，start要判断变化情况（同时，max_len要记录）
        max_len = 0

        start = 0
        end = 0
        for i, ch in enumerate(s):
            if ch in s[start:end]:
                # 二重序列在一重序列中的索引 + 某元素的二重序列索引 = 某元素的一重序列索引
                # 重复字符的下一个位置 -> +1
                start += s[start:end].index(ch) + 1
            end += 1

            if end - start > max_len:
                max_len = end - start
        
        return max_len

4. 寻找两个正序数组的中位数 (hard) 

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        # 偶数长度需要记录前一个元素，奇数长度直接返回当前元素
        m, n = len(nums1), len(nums2)
        total = m + n
        
        # 数组的指针：一个作为索引，另一个作为内容
        pre = cur = 0
        i = j = 0
        
        # 只需要遍历 total//2 + 1 次
        for _ in range(total // 2 + 1):
            pre = cur  # 保存上一个值
            # 移动较小元素的指针
            if i < m and (j >= n or nums1[i] < nums2[j]):
                cur = nums1[i]
                i += 1
            else:
                cur = nums2[j]
                j += 1
        
        # 根据总长度奇偶性返回结果
        if total % 2 == 1:
            return cur
        else:
            return (pre + cur) / 2.0

10. 正则表达式匹配 hard 

class Solution:
    def isMatch(self, s: str, p: str) -> bool:
        m, n = len(s), len(p)
        # 创建 (m+1) x (n+1) 的二维数组，初始值为 False
        dp = [[False] * (n + 1) for _ in range(m + 1)]
        dp[0][0] = True  # 空字符串匹配空模式
        
        # 初始化 s 为空字符串的情况
        for j in range(1, n + 1):
            if p[j-1] == '*':
                dp[0][j] = dp[0][j-2]
        
        # 填充 dp 数组
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if p[j-1] == s[i-1] or p[j-1] == '.':
                    dp[i][j] = dp[i-1][j-1]
                elif p[j-1] == '*':
                    # 先考虑匹配零个前面字符的情况
                    dp[i][j] = dp[i][j-2]
                    # 再考虑匹配一个或多个前面字符的情况
                    if p[j-2] == s[i-1] or p[j-2] == '.':
                        dp[i][j] |= dp[i-1][j]
        
        return dp[m][n]

11. 盛最多水的容器 

class Solution:
    def maxArea(self, height: List[int]) -> int:
        # 冒泡排序 -> 慢，超时
        # def get_V(i, j):
        #     w = abs(i-j)
        #     h = min(height[i], height[j])
        #     return w*h
        # max_v = 0
        # for i in range(len(height)):
        #     for j in range(i+1, len(height)):
        #         v = get_V(i, j)
        #         if v > max_v:
        #             max_v = v
        # return max_v

        # 从两边往中间前进（双指针） -> 快
        left = 0
        right = len(height) - 1
        max_area = 0
        
        while left < right:
            width = right - left
            h = min(height[left], height[right])
            current = width * h
            
            if current > max_area:
                max_area = current
            
            # 移动较矮的指针，试图找到更高的边
            if height[left] < height[right]:
                left += 1
            else:
                right -= 1
        
        return max_area

15. 三数之和 

class Solution:
    def threeSum(self, nums: list[int]) -> list[list[int]]:
        # 中间向两边(排序 + 双指针)
        res = []
        nums.sort()
        n = len(nums)
        for i in range(n):
            # 跳过i位置重复数字
            if i > 0 and nums[i] == nums[i-1]:
                continue
            l, r = i+1, n-1
            while l < r:
                s = nums[i] + nums[l] + nums[r]
                if s == 0:
                    res.append([nums[i], nums[l], nums[r]])
                    # 跳过左指针重复
                    while l < r and nums[l] == nums[l+1]:
                        l += 1
                    # 跳过右指针重复
                    while l < r and nums[r] == nums[r-1]:
                        r -= 1
                    l += 1
                    r -= 1
                elif s < 0:
                    l += 1
                else:
                    r -= 1
        return res

17. 电话号码的字母组合 

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        # 回溯：当不知道要套几个for循环时
        # 数字字母映射
        phone = {
            "2": ["a","b","c"],
            "3": ["d","e","f"],
            "4": ["g","h","i"],
            "5": ["j","k","l"],
            "6": ["m","n","o"],
            "7": ["p","q","r","s"],
            "8": ["t","u","v"],
            "9": ["w","x","y","z"]
        }
        res = []
        
        def backtrack(index, path):
            # 终止条件：路径长度等于数字总长时，写入结果（其他中间状态不写入）
            if index == len(digits):
                res.append(path)
                return
            # 遍历当前数字对应的字母表
            chars = phone[digits[index]]
            for c in chars:
                backtrack(index+1, path + c)
        
        backtrack(0, "")
        return res

19. 删除链表的倒数第 N 个结点 

class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        # 处理链表，双指针操作！！！
        
        # 由于可能会删除链表头部，用`哨兵节点`简化代码
        left = right = dummy = ListNode(next=head)

        # 右指针先向右走 n 步 （两指针相距n步）
        for _ in range(n): 
            right = right.next  

        # 左右指针一起走
        while right.next: # 右指针到底
            left = left.next
            right = right.next  
        
        # 左指针是倒数第 n+1 个节点
        # 左指针的下一个节点就是倒数第 n 个节点
        left.next = left.next.next  

        return dummy.next

20. 有效的括号 

class Solution:
    def isValid(self, s: str) -> bool:
        # 两两对应 -> 栈消除法

        def func(l, r):
            if l == '(' and r == ')':
                return True
            if l == '[' and r == ']':
                return True
            if l == '{' and r == '}':
                return True
            return False

        stack = []
        index = -1
        for ch in s:
            stack.append(ch)
            index += 1

            # 才一个元素, 直接下一个循环
            if index == 0:
                continue
            
            # 判断是否出栈
            flag = func(stack[index-1], stack[index])
            if flag:
                stack.pop(index)
                stack.pop(index-1)
                index -= 2

        # 结果
        return len(stack) == 0
