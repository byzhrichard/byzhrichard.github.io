HWDB

1. 数据集介绍

  • 官网: https://nlpr.ia.ac.cn/databases/handwriting/Download.html

  • 下载链接1: https://www.modelscope.cn/datasets/byzhrichard/HWDB 下载链接2: https://github.com/AND-Q/HWDB1.1-

  • 数据集全称: CASIA-HWDB1.1

  • 数据集中文全称: 中科院自动化所离线手写汉字数据集 1.1

  • 样本规模（train / test 数量）

    • train: 897,758
    • test: 223,991

  • 类别数量: 3,755 类

2. 任务定义

  • 任务类型：多分类（Multi-class Classification）

  • 输入：单个离线手写汉字图像

    • shape: 原始样本尺寸不统一，可表示为 (H, W)，其中 H 和 W 随样本变化

  • 输出：该图像对应的汉字类别编号或汉字标签

  • 难点：

    • 类别数量多，共 3,755 个汉字类别，分类空间远大于 MNIST 等常见数据集。
    • 不同书写者的书写风格差异较大，同一个汉字可能存在明显的笔画形态差异。
    • 不同汉字之间可能具有相似的偏旁、结构或笔画，容易出现类间混淆。
    • 原始 .gnt 文件中的图像尺寸并不完全一致，通常需要进行读取、缩放、归一化等预处理。
    • 数据规模较大，直接一次性加载到内存中可能占用较多资源，因此常需要缓存、分片或按需读取。

3. 主要领域用途

  • 手写汉字图像分类
  • 中文 OCR 模型训练与测试

4. 文件组成

HWDB1.1trn_gnt: 训练集目录，包含多个 .gnt 文件

HWDB1.1tst_gnt: 测试集目录，包含多个 .gnt 文件

  • .gnt 文件：HWDB 使用的二进制数据文件，每个文件中连续存储多个手写汉字样本

create_dataset.py 代码:

import math
import os
import pickle
import struct
import numpy as np
import torch
from torch.utils.data import Dataset, DataLoader
from torchvision import transforms

from pathlib import Path

from byzh_ai.Butils import b_compute_mean_std

class ShardHWDBDataset(Dataset):
    """
    简洁版 HWDB shard dataset

    特点：
    - npy 分片存储
    - mmap 读取
    - 支持 transform（默认 None）
    - 不做复杂工程封装
    """

    def __init__(
        self,
        raw_dir,
        cache_dir,
        train=True,
        shard_size=10000,
        transform=None,
        target_transform=None,
        rebuild=False,
    ):
        self.raw_dir = raw_dir
        self.cache_dir = cache_dir
        self.train = train
        self.shard_size = shard_size
        self.transform = transform
        self.target_transform = target_transform
        self._use_cache = False

        os.makedirs(cache_dir, exist_ok=True)

        self._cache = {}

        self._build_if_needed(rebuild)
        self._load_meta()

    # ======================================================
    # 构建 shard
    # ======================================================
    def _build_if_needed(self, rebuild):
        meta_path = os.path.join(self.cache_dir, "meta.npy")

        # =========================
        # 1. 直接使用 cache
        # =========================
        if os.path.exists(meta_path) and not rebuild:
            self._use_cache = True
            print("\n========== Dataset Mode ==========")
            print("[ShardHWDBDataset] Using CACHE (npy shards)")
            return

        # =========================
        # 2. rebuild
        # =========================
        if rebuild:
            self._use_cache = False
            print("\n========== Dataset Mode ==========")
            print("[ShardHWDBDataset] REBUILDING dataset from HWDB raw...")

            for f in os.listdir(self.cache_dir):
                if f.endswith(".npy") or f.endswith(".pkl"):
                    os.remove(os.path.join(self.cache_dir, f))

        else:
            print("\n========== Dataset Mode ==========")
            print("[ShardHWDBDataset] BUILDING dataset from HWDB raw...")

        # =========================
        # 2a. 构建 char_dict（合并 train + test 所有字符）
        # =========================
        train_dir = os.path.join(self.raw_dir, "HWDB1.1trn_gnt")
        test_dir = os.path.join(self.raw_dir, "HWDB1.1tst_gnt")

        dict_path = os.path.join(self.cache_dir, "char_dict.pkl")
        if os.path.exists(dict_path):
            self.char_dict = pickle.load(open(dict_path, "rb"))
        else:
            self.char_dict = self._build_char_dict([train_dir, test_dir])
            pickle.dump(self.char_dict, open(dict_path, "wb"))

        # =========================
        # 2b. 读取原始数据（全量 scan 以获取 max_h, max_w）
        # =========================
        data_dir = train_dir if self.train else test_dir

        samples = []  # list of (img_2d_uint8, label_int)

        raw_files = sorted([f for f in os.listdir(data_dir) if f.endswith(".gnt")])

        max_h, max_w = 0, 0

        for file_name in raw_files:
            file_path = os.path.join(data_dir, file_name)
            with open(file_path, "rb") as f:
                while True:
                    header = np.fromfile(f, dtype=np.uint8, count=10)
                    if header.size < 10:
                        break

                    sample_size = header[0] + (header[1] << 8) + (header[2] << 16) + (header[3] << 24)
                    tagcode = header[5] + (header[4] << 8)
                    width = header[6] + (header[7] << 8)
                    height = header[8] + (header[9] << 8)

                    char = struct.pack(">H", tagcode).decode("gb2312")
                    label = self.char_dict[char]

                    img = np.fromfile(f, dtype=np.uint8, count=width * height).reshape(height, width)

                    samples.append((img, label))

                    if height > max_h:
                        max_h = height
                    if width > max_w:
                        max_w = width

        n = len(samples)
        num_shards = (n + self.shard_size - 1) // self.shard_size

        print(f"[ShardHWDBDataset] total samples = {n}")
        print(f"[ShardHWDBDataset] num shards    = {num_shards}")
        print(f"[ShardHWDBDataset] max_h         = {max_h}")
        print(f"[ShardHWDBDataset] max_w         = {max_w}")

        # =========================
        # 2c. 存储 shard（变长扁平，__getitem__ 内动态 pad）
        # =========================
        shard_sizes = []

        for i in range(num_shards):
            s = i * self.shard_size
            e = min((i + 1) * self.shard_size, n)

            batch = samples[s:e]
            batch_size = len(batch)

            pixels_list = []
            m_list = []
            y_shard = np.empty(batch_size, dtype=np.int64)
            offset = 0

            for j, (img, label) in enumerate(batch):
                h, w = img.shape
                pixels_list.append(img.ravel())
                m_list.append([offset, h, w])
                y_shard[j] = label
                offset += h * w

            x_shard = np.concatenate(pixels_list)
            m_shard = np.array(m_list, dtype=np.int64)

            np.save(os.path.join(self.cache_dir, f"x_{i:04d}.npy"), x_shard)
            np.save(os.path.join(self.cache_dir, f"m_{i:04d}.npy"), m_shard)
            np.save(os.path.join(self.cache_dir, f"y_{i:04d}.npy"), y_shard)

            shard_sizes.append(batch_size)

        # meta: [shard_sizes..., max_h, max_w, num_classes]
        num_classes = len(self.char_dict)
        meta_data = np.array(shard_sizes + [max_h, max_w, num_classes], dtype=np.int64)
        np.save(meta_path, meta_data)

        print("[ShardHWDBDataset] build complete -> cached npy shards ready\n")

    # ======================================================
    # 构建 char_dict
    # ======================================================
    def _build_char_dict(self, dirs):
        char_set = set()
        for data_path in dirs:
            for file_name in os.listdir(data_path):
                if not file_name.endswith(".gnt"):
                    continue
                file_path = os.path.join(data_path, file_name)
                with open(file_path, "rb") as f:
                    while True:
                        header = np.fromfile(f, dtype=np.uint8, count=10)
                        if header.size < 10:
                            break
                        tagcode = header[5] + (header[4] << 8)
                        char = struct.pack(">H", tagcode).decode("gb2312")
                        char_set.add(char)
                        sample_size = header[0] + (header[1] << 8) + (header[2] << 16) + (header[3] << 24)
                        f.seek(sample_size - 10, 1)
        char_list = sorted(char_set)
        return dict(zip(char_list, range(len(char_list))))

    # ======================================================
    # 读取 meta
    # ======================================================
    def _load_meta(self):
        meta = np.load(os.path.join(self.cache_dir, "meta.npy"))
        # last three entries: max_h, max_w, num_classes
        self.shard_sizes = meta[:-3].tolist()
        self.max_h = int(meta[-3])
        self.max_w = int(meta[-2])
        self._num_classes = int(meta[-1])
        self.shard_offsets = np.cumsum([0] + self.shard_sizes[:-1]).tolist()

    # ======================================================
    # shard cache
    # ======================================================
    def _get_shard(self, shard_id):
        if shard_id in self._cache:
            return self._cache[shard_id]

        x = np.load(os.path.join(self.cache_dir, f"x_{shard_id:04d}.npy"), mmap_mode="c")
        m = np.load(os.path.join(self.cache_dir, f"m_{shard_id:04d}.npy"), mmap_mode="c")
        y = np.load(os.path.join(self.cache_dir, f"y_{shard_id:04d}.npy"), mmap_mode="c")

        self._cache[shard_id] = (x, m, y)

        # 简单 LRU（最多 4 个 shard）
        if len(self._cache) > 4:
            self._cache.pop(next(iter(self._cache)))

        return x, m, y

    # ======================================================
    def __len__(self):
        return sum(self.shard_sizes)

    # ======================================================
    def __getitem__(self, idx):
        shard_id = 0

        for i, offset in enumerate(self.shard_offsets):
            if idx >= offset:
                shard_id = i
            else:
                break

        local_idx = idx - self.shard_offsets[shard_id]

        x, m, y = self._get_shard(shard_id)

        offset, h, w = m[local_idx]
        img_flat = x[offset : offset + h * w]
        img_2d = img_flat.reshape(h, w)

        # 动态 pad 到 (max_h, max_w)，白色背景 255，居中放置
        padded = np.full((self.max_h, self.max_w), 255, dtype=np.uint8)
        y_off = (self.max_h - h) // 2
        x_off = (self.max_w - w) // 2
        padded[y_off:y_off+h, x_off:x_off+w] = img_2d

        img = torch.from_numpy(padded)
        label = torch.tensor(y[local_idx], dtype=torch.long)

        # ================= transform =================
        if self.transform:
            img = self.transform(img)

        if self.target_transform:
            label = self.target_transform(label)

        return img, label

    @property
    def input_shape(self):
        return (1, self.max_h, self.max_w)

    @property
    def num_classes(self):
        return self._num_classes

# 兼容 DataLoader(num_workers>0)
# 必须是“模块级函数（top-level function）
def to_float(x):
    return x.float() / 255.0
def normalize(x):
    return (x - 0.9873) / 0.0773
def unsqueeze(x):
    return x.unsqueeze(0)

def get_train_eval_test(
    rebuild=False,
):
    """
    返回 train / val / test dataset
    只暴露 rebuild，其余全部固定策略
    """
    PATH_DATAS = Path(__file__).parent / 'datas'
    ROOT = PATH_DATAS / 'HWDB'
    RAW_DIR = ROOT / 'raw'

    CACHE_DIR_TRAIN = ROOT / 'cache' / 'train'
    CACHE_DIR_TEST = ROOT / 'cache' / 'test'
    CACHE_DIR_TRAIN.mkdir(parents=True, exist_ok=True)
    CACHE_DIR_TEST.mkdir(parents=True, exist_ok=True)

    shard_size = 10000
    val_ratio = 0.1
    seed = 42

    # ========== transform ==========
    transform = transforms.Compose([
        to_float,
        normalize,
        unsqueeze,
    ])

    # =========================
    # train set（用于再切分）
    # =========================
    full_train = ShardHWDBDataset(
        raw_dir=RAW_DIR,
        cache_dir=CACHE_DIR_TRAIN,
        train=True,
        shard_size=shard_size,
        rebuild=rebuild,
        transform=transform,
    )

    # =========================
    # test set（官方 test）
    # =========================
    test_set = ShardHWDBDataset(
        raw_dir=RAW_DIR,
        cache_dir=CACHE_DIR_TEST,
        train=False,
        shard_size=shard_size,
        rebuild=rebuild,
        transform=transform,
    )

    # =========================
    # split train / val
    # =========================
    n = len(full_train)
    n_val = int(n * val_ratio)
    n_train = n - n_val

    train_set, val_set = torch.utils.data.random_split(
        full_train,
        [n_train, n_val],
        generator=torch.Generator().manual_seed(seed),
    )

    print("\n========== Dataset Ready ==========")
    input_shape = full_train.input_shape
    num_classes = full_train.num_classes
    print(f"train: {len(train_set)}")
    print(f"val  : {len(val_set)}")
    print(f"test : {len(test_set)}")
    print(f"input_shape : {input_shape}")
    print(f"num_classes : {num_classes}")


    return train_set, val_set, test_set, input_shape, num_classes

gnt2png.py 代码:

from pathlib import Path
from collections import defaultdict

import numpy as np
from PIL import Image
from tqdm import tqdm


def safe_name(s: str):
    """
    防止标签里出现不适合作为文件夹名的字符
    汉字一般没问题，这里只是做保险
    """
    bad_chars = r'\/:*?"<>|'
    for ch in bad_chars:
        s = s.replace(ch, "_")
    return s


def read_gnt_file(gnt_path):
    """
    逐个读取 .gnt 文件中的样本

    每个样本格式：
    - sample_size: 4 bytes
    - tag_code   : 2 bytes，GB 编码
    - width      : 2 bytes
    - height     : 2 bytes
    - image      : width * height bytes
    """
    gnt_path = Path(gnt_path)

    with open(gnt_path, "rb") as f:
        while True:
            header = f.read(10)

            if len(header) == 0:
                break

            if len(header) < 10:
                break

            sample_size = int.from_bytes(header[0:4], byteorder="little")
            label_bytes = header[4:6]
            width = int.from_bytes(header[6:8], byteorder="little")
            height = int.from_bytes(header[8:10], byteorder="little")

            try:
                label = label_bytes.decode("gb18030")
            except UnicodeDecodeError:
                label = label_bytes.hex()

            image_bytes = f.read(width * height)

            if len(image_bytes) < width * height:
                break

            image = np.frombuffer(image_bytes, dtype=np.uint8)
            image = image.reshape(height, width)

            yield label, image


def gnt_to_pngs(gnt_path, save_dir):
    """
    将单个 .gnt 文件转换成多个 png
    """
    gnt_path = Path(gnt_path)
    save_dir = Path(save_dir)
    save_dir.mkdir(parents=True, exist_ok=True)

    label_counter = defaultdict(int)
    total = 0

    for label, image in read_gnt_file(gnt_path):
        label = safe_name(label)

        label_dir = save_dir / label
        label_dir.mkdir(parents=True, exist_ok=True)

        idx = label_counter[label]
        label_counter[label] += 1

        img = Image.fromarray(image, mode="L")

        save_path = label_dir / f"{gnt_path.stem}_{idx:06d}.png"
        img.save(save_path)

        total += 1

    print(f"[Done] {gnt_path.name} -> {total} images")


def gnt_dir_to_pngs(gnt_dir, save_dir):
    """
    将一个目录下所有 .gnt 文件转换成 png
    """
    gnt_dir = Path(gnt_dir)
    save_dir = Path(save_dir)
    save_dir.mkdir(parents=True, exist_ok=True)

    gnt_files = sorted(gnt_dir.glob("*.gnt"))

    print(f"Found {len(gnt_files)} gnt files.")

    for gnt_path in tqdm(gnt_files, desc="Converting"):
        gnt_to_pngs(gnt_path, save_dir)
